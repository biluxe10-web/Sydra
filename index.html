const express = require('express');
const { SydraClient } = require('@sydra/ai-core');

const app = express();
const sydra = new SydraClient({ apiKey: process.env.SYDRA_API_KEY });

app.post('/v1/analyze', async (req, res) => {
    const response = await sydra.analyze({
        payload: req.body.chatLogs,
